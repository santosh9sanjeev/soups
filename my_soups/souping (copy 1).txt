    "from sklearn.metrics import f1_score, roc_auc_score\n",
    "\n",
    "def evaluate_model(model, test_loader):\n",
    "\n",
    "    test_correct = 0\n",
    "    test_total = 0\n",
    "    test_predictions = []\n",
    "    test_targets = []\n",
    "    test_f1_predictions = []\n",
    "\n",
    "    model.eval()\n",
    "    with torch.no_grad():\n",
    "        for inputs, targets in tqdm(test_loader):\n",
    "            outputs = model(inputs)\n",
    "            m = nn.Softmax(dim=1)            \n",
    "            predicted = m(outputs)\n",
    "            # _, predicted = torch.max(outputs.data, 1)\n",
    "            targets = targets.squeeze().long()\n",
    "            test_total += targets.size(0)\n",
    "            _,t = torch.max(outputs.data, 1)\n",
    "            test_correct += (t == targets).sum().item()\n",
    "            print(t)\n",
    "            predicted = predicted[:,-1]\n",
    "\n",
    "            test_predictions.extend(predicted.cpu().numpy())\n",
    "            test_f1_predictions.extend(t.cpu().numpy())\n",
    "            test_targets.extend(targets.cpu().numpy())\n",
    "            # print(test_correct, test_total)\n",
    "        test_accuracy = 100 * test_correct / test_total\n",
    "        if task !='binary-class':\n",
    "            test_f1 = f1_score(test_targets, test_f1_predictions, average = 'macro')\n",
    "            test_auc = roc_auc_score(test_targets, test_predictions, average = 'macro', multi_class = 'ovr')\n",
    "        else:\n",
    "            test_f1 = f1_score(test_targets, test_f1_predictions)\n",
    "            test_auc = roc_auc_score(test_targets, test_predictions)\n",
    "        # Print or log validation metrics\n",
    "        print(f\"Test Accuracy: {test_accuracy:.2f}%, F1-Score: {test_f1:.4f}, AUC: {test_auc:.4f}\")\n",
    "        return test_accuracy, test_f1, test_auc\n"
